Elliot Log — Setup Guide 🍼

5 minutes to set up a shared baby tracker for both phones

Part 1: Set Up the Google Sheet Backend

1 Open your Google Sheet

Go to your Elliot Logging spreadsheet.

Important: You need to convert it to a native Google Sheet first. Go to File → Save as Google Sheets. This creates a new file — use that one going forward.
2 Open Apps Script

In the new Google Sheet, go to Extensions → Apps Script. This opens a code editor.

3 Replace the code

Delete everything in the editor and paste this code:

function doGet(e) { var lock = LockService.getScriptLock(); lock.waitLock(10000); try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('Log'); if (!sheet) { sheet = ss.getSheets()[0]; } var action = e.parameter.action || 'read'; if (action === 'read') { var lastRow = sheet.getLastRow(); var headerRow = 4; // headers are in row 4 if (lastRow <= headerRow) { return ContentService.createTextOutput( JSON.stringify({ success: true, entries: [] }) ).setMimeType(ContentService.MimeType.JSON); } var dataRange = sheet.getRange( headerRow + 1, 1, lastRow - headerRow, 7 ); var data = dataRange.getValues(); var entries = []; for (var i = 0; i < data.length; i++) { var row = data[i]; if (!row[0] && !row[1]) continue; // skip empty rows entries.push({ date: row[0] ? row[0].toString() : '', time: row[1] ? row[1].toString() : '', type: row[2] ? row[2].toString() : '', side: row[3] ? row[3].toString() : '', shield: row[4] ? row[4].toString() : '', amount: row[5] ? row[5].toString() : '', details: row[6] ? row[6].toString() : '' }); } return ContentService.createTextOutput( JSON.stringify({ success: true, entries: entries }) ).setMimeType(ContentService.MimeType.JSON); } // Add entry if (action === 'add') { var date = e.parameter.date || ''; var time = e.parameter.time || ''; var type = e.parameter.type || ''; var side = e.parameter.side || ''; var shield = e.parameter.shield || ''; var amount = e.parameter.amount || ''; var details = e.parameter.details || ''; sheet.appendRow([date, time, type, side, shield, amount, details]); return ContentService.createTextOutput( JSON.stringify({ success: true, message: 'Entry added' }) ).setMimeType(ContentService.MimeType.JSON); } // Delete entry — matches on all 7 fields for precise targeting if (action === 'delete') { var delDate = e.parameter.date || ''; var delTime = e.parameter.time || ''; var delType = e.parameter.type || ''; var delSide = e.parameter.side || ''; var delShield = e.parameter.shield || ''; var delAmount = e.parameter.amount || ''; var delDetails = e.parameter.details || ''; var lastRow = sheet.getLastRow(); var headerRow = 4; // Search from bottom up to find matching row for (var r = lastRow; r > headerRow; r--) { var rowData = sheet.getRange(r, 1, 1, 7).getValues()[0]; var rowTime = rowData[1] ? rowData[1].toString() : ''; var rowType = rowData[2] ? rowData[2].toString() : ''; var rowSide = rowData[3] ? rowData[3].toString() : ''; var rowShield = rowData[4] ? rowData[4].toString() : ''; var rowAmount = rowData[5] ? rowData[5].toString() : ''; var rowDetails = rowData[6] ? rowData[6].toString() : ''; if (rowTime === delTime && rowType === delType && rowSide === delSide && rowShield === delShield && rowAmount === delAmount && rowDetails === delDetails) { sheet.deleteRow(r); return ContentService.createTextOutput( JSON.stringify({ success: true, message: 'Entry deleted' }) ).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput( JSON.stringify({ success: false, message: 'Entry not found' }) ).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput( JSON.stringify({ success: false, message: 'Unknown action' }) ).setMimeType(ContentService.MimeType.JSON); } finally { lock.releaseLock(); } } function doPost(e) { return doGet(e); }
4 Deploy as a web app

Click Deploy → New deployment (top right). Then:

• Click the gear icon → select "Web app"

• Set "Execute as" to "Me"

• Set "Who has access" to "Anyone"

• Click Deploy

• Google will ask you to authorize — click through the prompts

• Copy the Web App URL — it looks like:
https://script.google.com/macros/s/AKfyc.../exec

Save this URL! You'll paste it into the app in the next step.

Part 2: Set Up the App on Your Phones

5 Open the Elliot Log app

Open the elliot-log-app.html file in Safari on your iPhone (you can AirDrop it or put it in iCloud Drive).

6 Connect to your Google Sheet

Tap the ⚙️ Setup button at the top and paste your Web App URL from Step 4. Tap Save & Test.

7 Add to Home Screen

In Safari, tap the Share button (square with arrow) → "Add to Home Screen". Name it "Elliot Log".

Repeat steps 5–7 on your wife's phone!

You're all set! 🎉

Both phones now read and write to the same Google Sheet.
Every entry either of you logs shows up for both of you.
